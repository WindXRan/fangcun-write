# Coze Skill 开发指南

## 第一步：安装 Coze CLI

```bash
npm install -g @coze/cli
coze auth login --oauth
```

## 第二步：创建项目结构

```bash
mkdir story-cover-coze
cd story-cover-coze
```

## 第三步：开发封面生成器 Skill

### 3.1 创建 skill 配置文件

创建 `skill.json`：
```json
{
  "name": "story-cover-generator",
  "description": "根据书名、作者名、题材风格自动生成专业级网文封面",
  "version": "1.0.0",
  "author": "your-name",
  "parameters": {
    "type": "object",
    "properties": {
      "book_title": {
        "type": "string",
        "description": "小说书名"
      },
      "author_name": {
        "type": "string",
        "description": "作者名（可选）"
      },
      "genre": {
        "type": "string",
        "description": "题材风格（古言/现言/玄幻/都市等）",
        "enum": ["古言", "现言", "玄幻", "都市", "仙侠", "悬疑", "科幻", "历史", "其他"]
      }
    },
    "required": ["book_title"]
  }
}
```

### 3.2 创建主逻辑文件

创建 `index.js`：
```javascript
const { OpenAI } = require('openai');

exports.handler = async (event, context) => {
  const { book_title, author_name, genre } = event;
  
  // 1. 分析题材风格
  const detectedGenre = genre || detectGenre(book_title);
  
  // 2. 生成封面 prompt
  const prompt = generateCoverPrompt(book_title, author_name, detectedGenre);
  
  // 3. 调用 GPT-Image-2 生成封面
  const coverUrl = await generateCover(prompt);
  
  return {
    success: true,
    cover_url: coverUrl,
    genre: detectedGenre,
    prompt: prompt
  };
};

function detectGenre(bookTitle) {
  // 根据书名关键词判断题材
  const genreKeywords = {
    '古言': ['妃', '皇后', '王爷', '将军', '公主', '太子', '重生', '穿越'],
    '现言': ['总裁', '豪门', '校草', '学霸', '医生', '律师', '甜宠'],
    '玄幻': ['修仙', '仙尊', '神帝', '武道', '灵根', '丹药', '法宝'],
    '都市': ['都市', '重生', '系统', '异能', '透视', '神医'],
    '仙侠': ['仙', '道', '剑', '丹', '符', '阵'],
    '悬疑': ['诡', '案', '谜', '鬼', '灵异', '探案'],
    '科幻': ['星际', '机甲', '未来', '太空', '机器人'],
    '历史': ['大明', '大清', '三国', '唐朝', '宋朝']
  };
  
  for (const [genre, keywords] of Object.entries(genreKeywords)) {
    if (keywords.some(keyword => bookTitle.includes(keyword))) {
      return genre;
    }
  }
  
  return '其他';
}

function generateCoverPrompt(bookTitle, authorName, genre) {
  const styleMap = {
    '古言': '中国水墨画风格，古典优雅，淡雅色调，古风元素',
    '现言': '现代都市风格，时尚简约，明亮色调，都市元素',
    '玄幻': '玄幻仙侠风格，华丽特效，神秘色调，仙侠元素',
    '都市': '都市现代风格，写实画风，暖色调，都市生活元素',
    '仙侠': '仙侠风格，飘逸灵动，仙气缭绕，仙侠元素',
    '悬疑': '悬疑暗黑风格，神秘氛围，暗色调，悬疑元素',
    '科幻': '科幻未来风格，科技感，冷色调，未来元素',
    '历史': '历史古风风格，厚重感，复古色调，历史元素',
    '其他': '通用网文风格，吸引眼球，色彩鲜明'
  };
  
  const style = styleMap[genre] || styleMap['其他'];
  
  let prompt = `设计一个专业级网文封面：
- 书名：${book_title}
- 风格：${style}
- 要求：吸引眼球，符合网文特点，适合在番茄小说、起点等平台展示`;
  
  if (authorName) {
    prompt += `\n- 作者：${authorName}`;
  }
  
  return prompt;
}

async function generateCover(prompt) {
  // 这里需要调用 GPT-Image-2 API
  // 实际实现需要配置 API Key
  const openai = new OpenAI({
    apiKey: process.env.OPENAI_API_KEY,
  });
  
  const response = await openai.images.generate({
    model: "gpt-image-2",
    prompt: prompt,
    n: 1,
    size: "1024x1024",
  });
  
  return response.data[0].url;
}
```
